package main  
import "fmt"  
func main() {  
    var orderID int = 1001  
    var customerName string = "Stan Smith"  
    var isDelivered bool = false  
    orderIDs := []int{}  
    orderIDs = append(orderIDs, 101, 102, 103)  
    fmt.Printf("Order ID: %v\nCustomer Name: %v\nIs Delivered: %v\nOrderIDs: %v\ncap(OrderIDs) = %v, len(OrderIDs) = %v\n", orderID, customerName, isDelivered, orderIDs, cap(orderIDs), len(orderIDs))  
    orderCount := map[string]int{  
       "Michael Jordan": 1, "James LeBron": 3, "Kobe Bryant": 2, "Bill Russell": 6,  
    }  
    fmt.Printf("Order Count: %v\n", orderCount)  
    isReadyToShip := orderCount["James LeBron"] > 2  
    fmt.Println("Is James LeBron ready to ship?", isReadyToShip)  
}
